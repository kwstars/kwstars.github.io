package main

import (
	"encoding/json"
	"fmt"
	"os"
)

type ColorGroup struct {s
	ID     int
	Name   string
	Colors []string
}

func main() {
	m := []*ColorGroup{{
		ID:     1,
		Name:   "a",
		Colors: nil,
	}, {
		ID:     2,
		Name:   "a",
		Colors: nil,
	}}

	b, err := json.Marshal(m)
	if err != nil {
		fmt.Println("error:", err)
	}
	os.Stdout.Write(b)
}

//output
[{"ID":1,"Name":"a","Colors":null},{"ID":2,"Name":"a","Colors":null}]

package main

import (
	"encoding/json"
	"fmt"
	"os"
)

type TestDemo struct {
	Id     int
	Name   string
	HobbyA []string
	HobbyB []string
}

func main() {
	m := TestDemo{
		HobbyB: nil,
	}

	b, err := json.Marshal(m)
	if err != nil {
		fmt.Println("error:", err)
	}
	os.Stdout.Write(b)
}

//go1.15 output  
{"Id":0,"Name":"","HobbyA":null,"HobbyB":null}
